15use super::{Aes, Neon, PMull, Sha256, Sha512, CAPS_STATIC};
16use crate::polyfill::cstr;
17
18// ```
19// $ rustc +1.61.0 --print cfg --target=aarch64-apple-ios | grep -E "neon|aes|sha|pmull"
20// target_feature="aes"
21// target_feature="neon"
22// target_feature="sha2"
23// $ rustc +1.61.0 --print cfg --target=aarch64-apple-darwin | grep -E "neon|aes|sha|pmull"
24// target_feature="aes"
25// target_feature="neon"
26// target_feature="sha2"
27// target_feature="sha3"
28// ```
29//
30// XXX/TODO(coverage)/TODO(size): aarch64-apple-darwin is statically guaranteed to have "sha3" but
31// other aarch64-apple-* targets require dynamic detection. Since we don't have test coverage for
32// the other targets yet, we wouldn't have a way of testing the dynamic detection if we statically
33// enabled `Sha512` for -darwin. So instead, temporarily, we statically ignore the static
34// availability of the feature on -darwin so that it runs the dynamic detection.
35pub const MIN_STATIC_FEATURES: u32 = Neon::mask() | Aes::mask() | Sha256::mask() | PMull::mask();
36pub const FORCE_DYNAMIC_DETECTION: u32 = !MIN_STATIC_FEATURES;
37
38// MSRV: Enforce 1.61.0 onaarch64-apple-*, in particular) prior to. Earlier
39// versions of Rust before did not report the AAarch64 CPU features correctly
40// for these targets. Cargo.toml specifies `rust-version` but versions before
41// Rust 1.56 don't know about it.
42#[allow(clippy::assertions_on_constants)]
43const _AARCH64_APPLE_TARGETS_EXPECTED_FEATURES: () =
44    assert!((CAPS_STATIC & MIN_STATIC_FEATURES) == MIN_STATIC_FEATURES);
45
46// Ensure we don't accidentally allow features statically beyond
47// `MIN_STATIC_FEATURES` so that dynamic detection is done uniformly for
48// all of these targets.
49#[allow(clippy::assertions_on_constants)]
50const _AARCH64_APPLE_DARWIN_TARGETS_EXPECTED_FEATURES: () =
51    assert!(CAPS_STATIC == MIN_STATIC_FEATURES);

53pub fn detect_features() -> u32 {
54    fn detect_feature(name: cstr::Ref) -> bool {
55        use crate::polyfill;
56        use core::mem;
57        use libc::{c_int, c_void};
58
59        let mut value: c_int = 0;
60        let mut len = mem::size_of_val(&value);
61        let value_ptr = polyfill::ptr::from_mut(&mut value).cast::<c_void>();
62        // SAFETY: `value_ptr` is a valid pointer to `value` and `len` is the size of `value`.
63        let rc = unsafe {
64            libc::sysctlbyname(name.as_ptr(), value_ptr, &mut len, core::ptr::null_mut(), 0)
65        };
66        // All the conditions are separated so we can observe them in code coverage.
67        if rc != 0 {
68            return false;
69        }
70        debug_assert_eq!(len, mem::size_of_val(&value));
71        if len != mem::size_of_val(&value) {
72            return false;
73        }
74        value != 0
75    }
76
77    // We do not need to check for the presence of NEON, as Armv8-A always has it
78    const _ASSERT_NEON_DETECTED: () = assert!((CAPS_STATIC & Neon::mask()) == Neon::mask());
79
80    let mut features = 0;
81
82    // TODO(MSRV 1.77): Use c"..." literal.
83    const SHA512_NAME: cstr::Ref =
84        cstr::unwrap_const_from_bytes_with_nul(b"hw.optional.armv8_2_sha512\0");
85    if detect_feature(SHA512_NAME) {
86        features |= Sha512::mask();
87    }
88
89    features
90}

99        // We intentionally disable static feature detection for SHA-512.

102        if cfg!(target_os = "macos") {
103            use crate::cpu::{arm::Sha512, GetFeature as _};

105            // All aarch64-apple-darwin targets have SHA3 enabled statically...
106            assert!(cfg!(target_feature = "sha3"));
107
108            // ...so we should detect it.
109            let cpu = cpu::features();

111        }
