17use super::super::super::{inout::AliasingSlices3 as _, n0::N0, LimbSliceError, MAX_LIMBS};
18use crate::{
19    c,
20    limb::Limb,
21    polyfill::slice::{AsChunks, AsChunksMut},
22};
23use core::num::NonZeroUsize;
24
25#[inline]
26pub(in super::super::super) fn sqr_mont5(
27    mut in_out: AsChunksMut<Limb, 8>,
28    n: AsChunks<Limb, 8>,
29    n0: &N0,
30) -> Result<(), LimbSliceError> {
31    prefixed_extern! {
32        // `r` and/or 'a' may alias.
33        // XXX: BoringSSL (kinda, implicitly) declares this to return `int`.
34        // `num` must be a non-zero multiple of 8.
35        fn bn_sqr8x_mont(
36            rp: *mut Limb,
37            ap: *const Limb,
38            ap_again: *const Limb,
39            np: *const Limb,
40            n0: &N0,
41            num: c::NonZero_size_t);
42    }
43
44    let in_out = in_out.as_flattened_mut();
45    let n = n.as_flattened();
46    let num_limbs = NonZeroUsize::new(n.len()).ok_or_else(|| LimbSliceError::too_short(n.len()))?;
47
48    // Avoid stack overflow from the alloca inside.
49    if num_limbs.get() > MAX_LIMBS {
50        return Err(LimbSliceError::too_long(num_limbs.get()));
51    }
52
53    in_out
54        .with_non_dangling_non_null_pointers_rab(num_limbs, |r, a, a_again| {
55            let n = n.as_ptr(); // Non-dangling because num_limbs > 0.
56            unsafe { bn_sqr8x_mont(r, a, a_again, n, n0, num_limbs) };
57        })
58        .map_err(LimbSliceError::len_mismatch)
59}
