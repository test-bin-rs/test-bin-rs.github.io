57#[derive(Clone)]
58pub(crate) enum RepoRef {
59    ByOwnerAndName(String, String),
60    ById(RepositoryId),
61}
62
63impl std::fmt::Display for RepoRef {
64    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
65        match self {
66            RepoRef::ByOwnerAndName(owner, name) => write!(f, "repos/{owner}/{name}"),
67            RepoRef::ById(id) => write!(f, "repositories/{id}"),
68        }
69    }
70}

187    /// ```no_run
188    /// # async fn run() -> octocrab::Result<()> {
189    /// # let master_sha = "";

205        let route = format!("/{}/git/refs", self.repo);

214            .await
215    }

679        path: impl AsRef<str>,
680    ) -> Result<http::Response<BoxBody<Bytes, crate::Error>>> {
681        let route = format!(
682            "/{repo}/contents/{path}",
683            repo = self.repo,
684            path = path.as_ref(),
685        );
686
687        let uri = self
688            .crab
689            .parameterized_uri(route, Some(&[("ref", &reference.into().0)]))?;
690        let request = Builder::new()
691            .uri(uri)
692            .method(http::Method::GET)
693            .header(ACCEPT, "application/vnd.github.v3.raw");
694        self.crab
695            .execute(self.crab.build_request(request, None::<&()>)?)
696            .await
697    }
